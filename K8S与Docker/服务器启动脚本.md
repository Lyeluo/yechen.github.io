## 启动jar包的脚本
```bash
#!/bin/bash

ps -ef|grep app|grep -v grep|cut -c 9-15|xargs kill -9
#后台启动jar，不输出日志
nohup java -Xms128m -Xmx512m -jar app.jar >/dev/null 2>&1 &
#循环判断是否启动，后续可以改进为判断heathz接口是否返回200
sleep 2
i=1
while [ $i -lt 10000 ]
do
  num=$(ps -ef|grep xxl-job-admin|grep -v grep|cut -c 9-15)
  #echo $num
  if [[ -n "$num" ]]
  then
    i=10001
    echo 'app started'
  else
    let i++
  fi
done
#设置一个空的打印语句，防止java进程挂掉时，容器挂掉
tail -f /dev/null
```
